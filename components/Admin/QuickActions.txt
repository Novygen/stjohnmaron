"use client";

import React from "react";

interface QuickActionsProps {
  onEdit: () => void;
  onStatus: () => void;
}

export default function QuickActions({ onEdit, onStatus }: QuickActionsProps) {

  return (
    <div className="relative inline-block text-left z-50">
      <button onClick={() => { onEdit(); }} className="block w-full text-left px-4 py-2 hover:bg-gray-100">
        Edit
      </button>
      <button onClick={() => { onStatus(); }} className="block w-full text-left px-4 py-2 hover:bg-gray-100">
        Status
      </button>
    </div>
  );
}
